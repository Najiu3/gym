import requests from bs4 import BeautifulSoup import json def fetch_staff(): url = "https://www.cardinalsoccerclub.org/about/our-staff/15999" response = requests.get(url) response.raise_for_status() soup = BeautifulSoup(response.text, "html.parser") staff_data = {} # Go through each header (assumed to be h2 tags for sections) for header in soup.find_all("h2"): section_title = header.get_text(strip=True) section_list = [] for sibling in header.find_next_siblings(): if sibling.name == "h2": break # End of this section if sibling.name in ["p", "div"]: text = sibling.get_text(separator="\n", strip=True) for line in text.split("\n"): if line: section_list.append(line) if section_list: staff_data[section_title] = section_list return staff_data def save_to_json(data, filename="cardinal_staff.json"): with open(filename, "w", encoding="utf-8") as f: json.dump(data, f, indent=2, ensure_ascii=False) print(f"✅ Staff data saved to {filename}") if __name__ == "__main__": try: staff = fetch_staff() save_to_json(staff) except Exception as e: print("❌ Error fetching staff data:", e)
